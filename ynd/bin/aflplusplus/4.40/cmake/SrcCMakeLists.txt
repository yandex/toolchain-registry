cmake_minimum_required(VERSION 3.20)

if(NOT DEFINED LLVM_VERSION_MAJOR)
  message(FATAL_ERROR "LLVM_VERSION_MAJOR is not set. This must be built as part of LLVM.")
endif()

if(LLVM_VERSION_MAJOR LESS 12)
  message(FATAL_ERROR "AFL++ LTO requires LLVM >= 12, got ${LLVM_VERSION_MAJOR}")
endif()

set(AFL_VERSION "++4.40c")
set(AFL_PATH "${CMAKE_INSTALL_PREFIX}/lib/afl" CACHE STRING "")
set(AFL_BIN_PATH "${CMAKE_INSTALL_PREFIX}/bin" CACHE STRING "")
set(AFL_INCLUDE_PATH "${CMAKE_INSTALL_PREFIX}/include/afl" CACHE STRING "")
set(AFL_LLVM_BINDIR "${CMAKE_INSTALL_PREFIX}/bin")
set(AFL_LLVM_LIBDIR "${CMAKE_INSTALL_PREFIX}/lib")
set(AFL_CLANG_BIN "${AFL_LLVM_BINDIR}/clang")
set(AFL_CLANGPP_BIN "${AFL_LLVM_BINDIR}/clang++")

set(AFL_SRC_DIR "${CMAKE_CURRENT_SOURCE_DIR}/..")

foreach(_src
  "${AFL_SRC_DIR}/src/afl-cc.c"
  "${AFL_SRC_DIR}/src/afl-common.c"
)
  if(NOT EXISTS "${_src}")
    message(FATAL_ERROR "Required AFL++ source file not found: ${_src}")
  endif()
endforeach()

# Common compile options and includes for all AFL++ targets
set(AFL_COMMON_COMPILE_OPTIONS
  -O3 -funroll-loops -Wall -Wno-unused-function -Wno-deprecated
  -Wno-pointer-sign -Wno-cast-qual -Wno-variadic-macros
)

set(AFL_COMMON_INCLUDES
  ${AFL_SRC_DIR}/include
  ${AFL_SRC_DIR}/instrumentation
)

set(AFL_COMMON_DEFINITIONS
  AFL_PATH="${AFL_PATH}"
  BIN_PATH="${AFL_BIN_PATH}"
  DOC_PATH="${CMAKE_INSTALL_PREFIX}/share/doc/afl"
  VERSION="${AFL_VERSION}"
)

add_executable(afl-cc
  ${AFL_SRC_DIR}/src/afl-cc.c
  ${AFL_SRC_DIR}/src/afl-common.c
)

target_include_directories(afl-cc PRIVATE ${AFL_COMMON_INCLUDES})

target_compile_definitions(afl-cc PRIVATE
  AFL_STATIC_LLVM_PASS=1
  ${AFL_COMMON_DEFINITIONS}
  AFL_INCLUDE_PATH="${AFL_INCLUDE_PATH}"
  LLVM_BINDIR="${AFL_LLVM_BINDIR}"
  LLVM_LIBDIR="${AFL_LLVM_LIBDIR}"
  LLVM_VERSION="${LLVM_VERSION_MAJOR}.${LLVM_VERSION_MINOR}.${LLVM_VERSION_PATCH}"
  LLVM_MAJOR=${LLVM_VERSION_MAJOR}
  LLVM_MINOR=${LLVM_VERSION_MINOR}
  CLANG_BIN="${AFL_CLANG_BIN}"
  CLANGPP_BIN="${AFL_CLANGPP_BIN}"
  USE_BINDIR=1
  CFLAGS_OPT=""
  AFL_CLANG_FLTO="-flto=full"
  AFL_REAL_LD=""
  AFL_CLANG_LDPATH=""
  AFL_CLANG_FUSELD=""
)

target_compile_options(afl-cc PRIVATE ${AFL_COMMON_COMPILE_OPTIONS})
target_link_libraries(afl-cc PRIVATE m)

install(TARGETS afl-cc
  RUNTIME DESTINATION bin
  COMPONENT afl-cc
)

add_custom_target(install-afl-cc
  DEPENDS afl-cc
  COMMAND ${CMAKE_COMMAND} -DCMAKE_INSTALL_COMPONENT=afl-cc
    -P ${CMAKE_BINARY_DIR}/cmake_install.cmake
)

install(FILES
  ${AFL_SRC_DIR}/dynamic_list.txt
  DESTINATION lib/afl
  COMPONENT afl-cc
)

add_library(afl-common-objs OBJECT
  ${AFL_SRC_DIR}/src/afl-common.c
  ${AFL_SRC_DIR}/src/afl-sharedmem.c
  ${AFL_SRC_DIR}/src/afl-forkserver.c
  ${AFL_SRC_DIR}/src/afl-performance.c
)

target_include_directories(afl-common-objs PRIVATE ${AFL_COMMON_INCLUDES})
target_compile_definitions(afl-common-objs PRIVATE ${AFL_COMMON_DEFINITIONS})
target_compile_options(afl-common-objs PRIVATE ${AFL_COMMON_COMPILE_OPTIONS})

# Mutator/python objects needed by afl-fuzz, afl-showmap, afl-tmin
add_library(afl-mutator-objs OBJECT
  ${AFL_SRC_DIR}/src/afl-fuzz-mutators.c
  ${AFL_SRC_DIR}/src/afl-fuzz-python.c
)

target_include_directories(afl-mutator-objs PRIVATE ${AFL_COMMON_INCLUDES})
target_compile_definitions(afl-mutator-objs PRIVATE ${AFL_COMMON_DEFINITIONS})
target_compile_options(afl-mutator-objs PRIVATE ${AFL_COMMON_COMPILE_OPTIONS})

add_executable(afl-fuzz
  ${AFL_SRC_DIR}/src/afl-fuzz.c
  ${AFL_SRC_DIR}/src/afl-fuzz-bitmap.c
  ${AFL_SRC_DIR}/src/afl-fuzz-cmplog.c
  ${AFL_SRC_DIR}/src/afl-fuzz-extras.c
  ${AFL_SRC_DIR}/src/afl-fuzz-frameshift.c
  ${AFL_SRC_DIR}/src/afl-fuzz-ijon.c
  ${AFL_SRC_DIR}/src/afl-fuzz-init.c
  ${AFL_SRC_DIR}/src/afl-fuzz-mutators.c
  ${AFL_SRC_DIR}/src/afl-fuzz-one.c
  ${AFL_SRC_DIR}/src/afl-fuzz-python.c
  ${AFL_SRC_DIR}/src/afl-fuzz-queue.c
  ${AFL_SRC_DIR}/src/afl-fuzz-redqueen.c
  ${AFL_SRC_DIR}/src/afl-fuzz-run.c
  ${AFL_SRC_DIR}/src/afl-fuzz-sanfuzz.c
  ${AFL_SRC_DIR}/src/afl-fuzz-skipdet.c
  ${AFL_SRC_DIR}/src/afl-fuzz-state.c
  ${AFL_SRC_DIR}/src/afl-fuzz-stats.c
  ${AFL_SRC_DIR}/src/afl-fuzz-statsd.c
  $<TARGET_OBJECTS:afl-common-objs>
)

target_include_directories(afl-fuzz PRIVATE ${AFL_COMMON_INCLUDES})
target_compile_definitions(afl-fuzz PRIVATE ${AFL_COMMON_DEFINITIONS})
target_compile_options(afl-fuzz PRIVATE ${AFL_COMMON_COMPILE_OPTIONS})
target_link_libraries(afl-fuzz PRIVATE m dl rt)

add_executable(afl-showmap
  ${AFL_SRC_DIR}/src/afl-showmap.c
  $<TARGET_OBJECTS:afl-common-objs>
  $<TARGET_OBJECTS:afl-mutator-objs>
)

target_include_directories(afl-showmap PRIVATE ${AFL_COMMON_INCLUDES})
target_compile_definitions(afl-showmap PRIVATE ${AFL_COMMON_DEFINITIONS})

add_executable(afl-tmin
  ${AFL_SRC_DIR}/src/afl-tmin.c
  $<TARGET_OBJECTS:afl-common-objs>
  $<TARGET_OBJECTS:afl-mutator-objs>
)

target_include_directories(afl-tmin PRIVATE ${AFL_COMMON_INCLUDES})
target_compile_definitions(afl-tmin PRIVATE ${AFL_COMMON_DEFINITIONS})

target_link_libraries(afl-tmin PRIVATE m dl rt)

  ${AFL_SRC_DIR}/src/afl-analyze.c
  $<TARGET_OBJECTS:afl-common-objs>
)

target_include_directories(afl-analyze PRIVATE ${AFL_COMMON_INCLUDES})

  ${AFL_SRC_DIR}/src/afl-gotcpu.c

foreach(_tool afl-fuzz afl-showmap afl-tmin afl-analyze afl-gotcpu)
  install(TARGETS ${_tool}
    RUNTIME DESTINATION bin
    COMPONENT ${_tool}
  )
  add_custom_target(install-${_tool}
    DEPENDS ${_tool}
    COMMAND ${CMAKE_COMMAND} -DCMAKE_INSTALL_COMPONENT=${_tool}
      -P ${CMAKE_BINARY_DIR}/cmake_install.cmake
  )
endforeach()

foreach(_script afl-cmin afl-cmin.bash afl-plot afl-whatsup afl-addseeds
                afl-persistent-config afl-system-config)
  if(EXISTS "${AFL_SRC_DIR}/${_script}")
    install(PROGRAMS ${AFL_SRC_DIR}/${_script}
      DESTINATION bin
      COMPONENT afl-fuzz
    )
  endif()
endforeach()
